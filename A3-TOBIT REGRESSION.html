# Define the log-likelihood function
def tobit_loglike(params, exog, endog, left=0):
    beta = params[:-1]
    sigma = params[-1]
    XB = np.dot(exog, beta)
    z = (endog - XB) / sigma
    ll = np.where(endog > left,
                  np.log(norm.pdf(z) / sigma),
                  np.log(norm.cdf(z)))
    return -np.sum(ll)
